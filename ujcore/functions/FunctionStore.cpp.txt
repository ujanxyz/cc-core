#include "ujcore/functions/FunctionStore.hpp"

namespace ujcore {

bool FunctionStore::AddFunction(FunctionSpec spec, std::string* out_error = nullptr) {
    specs_[spec.id] = spec;
    return true;
}

bool FunctionStore::HasFunction(std::string_view func_id) const {
    return specs_.contains(func_id);
}

const FunctionSpec* FunctionStore::Lookup(std::string_view func_id) const {
    if (specs_.contains(func_id)) {
        return &specs_.at(func_id);
    } else {
        return nullptr;
    }
}

}  // namespace ujcore
